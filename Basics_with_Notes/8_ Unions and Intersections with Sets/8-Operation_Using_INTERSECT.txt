(
	SELECT * FROM products
	ORDER BY price DESC
	LIMIT 4 -- we can't use ; while using INTERSECT keyword to intersection
)

INTERSECT -- it will print only common values and ignore duplicates

(
	SELECT * FROM products
	ORDER BY price/weight DESC
	LIMIT 4 -- we can't use ; while using INTERSECT keyword to intersection
); -- now can put ; here to mean that intersection is end




(
	SELECT * FROM products
	ORDER BY price DESC
	LIMIT 4 -- we can't use ; while using INTERSECT keyword to intersection
)

INTERSECT ALL-- it will print common values with duplicates

(
	SELECT * FROM products
	ORDER BY price/weight DESC
	LIMIT 4 -- we can't use ; while using INTERSECT keyword to intersection
); -- now can put ; here to mean that intersection is end
