SELECT user_id, MAX(id) -- here id means the unique comment ids
FROM comments
GROUP BY user_id



SELECT user_id, MIN(id) -- here id means the unique comment ids
FROM comments
GROUP BY user_id



SELECT user_id, COUNT(id) -- here id means the unique comment ids
FROM comments
GROUP BY user_id



SELECT user_id, AVG(id) -- here id means the unique comment ids
FROM comments
GROUP BY user_id



SELECT user_id, SUM(id) -- here id means the unique comment ids
FROM comments
GROUP BY user_id
